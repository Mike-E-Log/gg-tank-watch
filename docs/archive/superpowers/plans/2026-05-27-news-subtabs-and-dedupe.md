> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reorganize the News tab into two sub-tabs ("Updates" and "Timeline") and fix the inflated Articles count caused by duplicate ABC7 live-blog entries in `videos[]`.

**Architecture:** Pure-HTML/JS sub-tab switcher inside `#panel-news` (no router). Existing news content (situation banner, filter chips, `news-feed`) becomes the default "Updates" sub-panel; existing timeline content (header, filter chips, `timeline-feed`) becomes the "Timeline" sub-panel. In the Python writer, dedupe `videos[]` by canonicalized URL before persisting, and re-derive `is_video` from URL pattern as the authoritative source of truth.

**Tech Stack:** Vanilla JS + CSS in single-file `dashboard.html`; Python 3 standard library in `scripts/update_status.py`; pytest-style eval harness under `eval/`.

**Out of scope (deferred to a follow-up plan):** Fetching real publication timestamps from YouTube oEmbed and source `<meta property="article:published_time">`. This plan accepts the current rounded timestamps and only corrects classification + dedupe.

```python
"""Behavioral tests for video dedupe + URL-based reclassification.

Covers:
- Duplicate URLs in raw videos[] collapse to one entry in status.json
- is_video is re-derived from URL pattern (youtube.com or /video/ in path),
  overriding any incoming is_video flag
"""

from __future__ import annotations

import json
import shutil
import subprocess
import sys
from pathlib import Path

CATEGORY = "behavioral"

REPO_ROOT = Path(__file__).resolve().parent.parent
WRITER = REPO_ROOT / "scripts" / "update_status.py"
SANDBOX = Path(__file__).resolve().parent / ".last_run" / "videos_dedupe"


def _reset_sandbox():
    if SANDBOX.exists():
        shutil.rmtree(SANDBOX)
    SANDBOX.mkdir(parents=True)
    (SANDBOX / "config.json").write_text(json.dumps({
        "zone_status": "outside_downwind",
        "writer_interval_minutes": 30,
        "incident": {"name": "Test", "kind": "test"},
    }))


def _run_writer(facts):
    (SANDBOX / "facts.json").write_text(json.dumps(facts))
    result = subprocess.run(
        [sys.executable, str(WRITER),
         "--config", str(SANDBOX / "config.json"),
         "--facts", str(SANDBOX / "facts.json"),
         "--out", str(SANDBOX / "status.json")],
        capture_output=True, text=True,
    )
    assert result.returncode == 0, result.stderr
    return json.loads((SANDBOX / "status.json").read_text())


def test_duplicate_urls_collapse():
    _reset_sandbox()
    snap = _run_writer({"videos": [
        {"outlet": "ABC7", "title": "A", "url": "https://abc7.com/live-updates/x"},
        {"outlet": "ABC7", "title": "B", "url": "https://abc7.com/live-updates/x"},
        {"outlet": "ABC7", "title": "C", "url": "https://abc7.com/live-updates/x"},
        {"outlet": "NBC",  "title": "D", "url": "https://nbclosangeles.com/live"},
    ]})
    urls = [v["url"] for v in snap["videos"]]
    assert urls.count("https://abc7.com/live-updates/x") == 1, urls
    assert len(snap["videos"]) == 2, snap["videos"]


def test_url_pattern_overrides_is_video_flag():
    _reset_sandbox()
    snap = _run_writer({"videos": [
        {"outlet": "ABC7", "title": "story",   "url": "https://abc7.com/story/123", "is_video": True},
        {"outlet": "NBC",  "title": "explain", "url": "https://nbclosangeles.com/video/news/local/x/", "is_video": False},
        {"outlet": "YT",   "title": "clip",    "url": "https://www.youtube.com/watch?v=abc123"},
    ]})
    by_url = {v["url"]: v for v in snap["videos"]}
    assert by_url["https://abc7.com/story/123"]["is_video"] is False
    assert by_url["https://nbclosangeles.com/video/news/local/x/"]["is_video"] is True
    assert by_url["https://www.youtube.com/watch?v=abc123"]["is_video"] is True


def test_dedupe_keeps_first_occurrence():
    _reset_sandbox()
    snap = _run_writer({"videos": [
        {"outlet": "ABC7", "title": "FIRST",  "url": "https://abc7.com/x"},
        {"outlet": "ABC7", "title": "SECOND", "url": "https://abc7.com/x"},
    ]})
    assert len(snap["videos"]) == 1
    assert snap["videos"][0]["title"] == "FIRST"
```

Replace it with:

```python
    # Videos: list of {outlet, title, url, thumbnail_url?, published_iso?, youtube_id?}
    # - Dedupe by URL (keep first occurrence) to prevent live-blog URLs being
    #   duplicated across multiple "stories" that all point at the same page.
    # - is_video is re-derived from URL pattern; any incoming flag is ignored
    #   so display classification stays consistent with the actual destination.
    # - If youtube_id is set and thumbnail_url is missing, derive YouTube hqdefault.
    raw_videos = facts.get("videos") if "videos" in facts else prev.get("videos") or []
    videos = []
    seen_urls = set()
    for v in raw_videos or []:
        v = dict(v)
        url = (v.get("url") or "").strip()
        if url:
            if url in seen_urls:
                continue
            seen_urls.add(url)
        if v.get("youtube_id") or "youtube.com/" in url or "youtu.be/" in url or "/video/" in url:
            v["is_video"] = True
        else:
            v["is_video"] = False
        if not v.get("thumbnail_url") and v.get("youtube_id"):
            v["thumbnail_url"] = f"https://img.youtube.com/vi/{v['youtube_id']}/hqdefault.jpg"
        videos.append(v)
```

```javascript
/* ============ NEWS SUB-TABS ============ */
function switchNewsSubtab(name) {
  var tabs = document.querySelectorAll(".news-subtab");
  var panels = document.querySelectorAll(".news-subpanel");
  for (var i = 0; i < tabs.length; i++) {
    var isActive = tabs[i].id === "news-subtab-" + name;
    tabs[i].classList.toggle("active", isActive);
    tabs[i].setAttribute("aria-selected", isActive ? "true" : "false");
  }
  for (var j = 0; j < panels.length; j++) {
    panels[j].classList.toggle("active", panels[j].id === "news-subpanel-" + name);
  }
}
window.switchNewsSubtab = switchNewsSubtab;

```
